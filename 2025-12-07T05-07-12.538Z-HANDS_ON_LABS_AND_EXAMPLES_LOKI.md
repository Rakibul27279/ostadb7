```python
#!/usr/bin/env python3
from flask import Flask, request, jsonify
import logging
import json
import time
import random
from datetime import datetime
from prometheus_client import Counter, Histogram, Gauge, generate_latest
import os

# Setup JSON logging
class JSONFormatter(logging.Formatter):
    def format(self, record):
        log_obj = {
            'timestamp': datetime.utcnow().isoformat() + 'Z',
            'level': record.levelname,
            'message': record.getMessage(),
            'module': record.module,
        }
        if hasattr(record, 'user_id'):
            log_obj['user_id'] = record.user_id
        if hasattr(record, 'duration_ms'):
            log_obj['duration_ms'] = record.duration_ms
        return json.dumps(log_obj)

handler = logging.StreamHandler()
handler.setFormatter(JSONFormatter())
logger = logging.getLogger()
logger.addHandler(handler)
logger.setLevel(logging.INFO)

app = Flask(__name__)

# Prometheus metrics
REQUEST_COUNT = Counter('http_requests_total', 'Total HTTP requests', ['method', 'endpoint', 'status'])
REQUEST_LATENCY = Histogram('http_request_duration_seconds', 'HTTP request latency', ['method', 'endpoint'])
ACTIVE_REQUESTS = Gauge('http_requests_active', 'Active HTTP requests')

@app.before_request
def before_request():
    request.start_time = time.time()
    ACTIVE_REQUESTS.inc()

@app.after_request
def after_request(response):
    duration = time.time() - request.start_time
    REQUEST_COUNT.labels(
        method=request.method,
        endpoint=request.endpoint or 'unknown',
        status=response.status_code
    ).inc()
    REQUEST_LATENCY.labels(
        method=request.method,
        endpoint=request.endpoint or 'unknown'
    ).observe(duration)
    ACTIVE_REQUESTS.dec()
    
    logger.info(f"{request.method} {request.path} - {response.status_code}", extra={
        'duration_ms': round(duration * 1000, 2)
    })
    return response

@app.route('/health')
def health():
    return jsonify({"status": "healthy"})

@app.route('/metrics')
def metrics():
    return generate_latest()

@app.route('/api/users/<user_id>')
def get_user(user_id):
    time.sleep(random.uniform(0.01, 0.1))
    if random.random() < 0.1:
        logger.error(f"Failed to fetch user {user_id}")
        return jsonify({"error": "Not found"}), 404
    return jsonify({"id": user_id, "name": f"User {user_id}"})

@app.route('/api/orders', methods=['POST'])
def create_order():
    data = request.get_json() or {}
    time.sleep(random.uniform(0.05, 0.2))
    if random.random() < 0.05:
        logger.error("Order creation failed")
        return jsonify({"error": "Failed"}), 500
    return jsonify({"order_id": f"ORD-{random.randint(1000,9999)}", "status": "created"}), 201

if __name__ == '__main__':
    logger.info("Python app starting on port 5000")
    app.run(host='0.0.0.0', port=5000)
```

`apps/nodejs-app/server.js`:
```javascript
const express = require('express');
const promClient = require('prom-client');
const winston = require('winston');

const app = express();
app.use(express.json());

// Setup Winston logger for JSON logs
const logger = winston.createLogger({
  format: winston.format.combine(
    winston.format.timestamp(),
    winston.format.json()
  ),
  transports: [
    new winston.transports.Console()
  ]
});

// Prometheus metrics
const register = new promClient.Registry();
promClient.collectDefaultMetrics({ register });

const httpRequestCounter = new promClient.Counter({
  name: 'http_requests_total',
  help: 'Total HTTP requests',
  labelNames: ['method', 'route', 'status_code'],
  registers: [register]
});

const httpRequestDuration = new promClient.Histogram({
  name: 'http_request_duration_seconds',
  help: 'HTTP request duration',
  labelNames: ['method', 'route'],
  registers: [register]
});

// Middleware
app.use((req, res, next) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    httpRequestCounter.labels(req.method, req.route?.path || req.path, res.statusCode).inc();
    httpRequestDuration.labels(req.method, req.route?.path || req.path).observe(duration);
    
    logger.info({
      method: req.method,
      path: req.path,
      status: res.statusCode,
      duration_ms: Math.round(duration * 1000)
    });
  });
  
  next();
});

// Routes
app.get('/health', (req, res) => {
  res.json({ status: 'healthy' });
});

app.get('/metrics', async (req, res) => {
  res.set('Content-Type', register.contentType);
  res.end(await register.metrics());
});

app.get('/api/data', (req, res) => {
  const delay = Math.random() * 100;
  setTimeout(() => {
    if (Math.random() < 0.1) {
      logger.error({ message: 'Data fetch failed', error: 'Database timeout' });
      res.status(500).json({ error: 'Internal error' });
    } else {
      res.json({ data: Array.from({length: 10}, (_, i) => ({ id: i, value: Math.random() })) });
    }
  }, delay);
});

const PORT = 3001;
app.listen(PORT, () => {
  logger.info({ message: `Node.js app started on port ${PORT}` });
});
```

### Logstash Pipeline

`config/logstash/pipeline/logstash.conf`:
```conf
input {
  tcp {
    port => 5000
    codec => json
  }
  
  beats {
    port => 5044
  }
}

filter {
  if [type] == "nginx-access" {
    grok {
      match => { 
        "message" => '%{IPORHOST:client_ip} - %{USER:ident} \[%{HTTPDATE:timestamp}\] "%{WORD:method} %{DATA:request} HTTP/%{NUMBER:http_version}" %{NUMBER:status_code} %{NUMBER:bytes_sent} "%{DATA:referrer}" "%{DATA:user_agent}"' 
      }
    }
    
    date {
      match => [ "timestamp", "dd/MMM/yyyy:HH:mm:ss Z" ]
      target => "@timestamp"
    }
    
    mutate {
      convert => {
        "status_code" => "integer"
        "bytes_sent" => "integer"
      }
    }
    
    geoip {
      source => "client_ip"
      target => "geoip"
    }
  }
  
  if [type] == "application-json" {
    json {
      source => "message"
      target => "app"
    }
    
    mutate {
      rename => {
        "[app][level]" => "log_level"
        "[app][message]" => "log_message"
      }
    }
  }
}

output {
  elasticsearch {
    hosts => ["elasticsearch:9200"]
    index => "logs-%{[type]}-%{+YYYY.MM.dd}"
  }
  
  stdout {
    codec => rubydebug
  }
}
```

    "File uploaded to S3"

  )

  # Create JSON log

{
  "timestamp": "$(date -u +%Y-%m-%dT%H:%M:%S.%3NZ)",
  "level": "$LEVEL",
  "service": "sample-app",
  "message": "$MESSAGE",
  "user_id": "user-$((RANDOM % 100))",
  "request_id": "req-$(uuidgen)",
  "duration_ms": $((RANDOM % 1000)),
  "type": "application-json"

  echo "$LOG"
  echo "$LOG" | nc localhost 5000
  
  sleep 1
done

## Lab 3: Security Monitoring (Simple SIEM)

### Python SIEM Script

`siem/simple_siem.py`:
```python
#!/usr/bin/env python3
"""
Simple SIEM (Security Information and Event Management) System
Monitors logs from Loki and detects security threats
"""

import requests
import time
from datetime import datetime, timedelta
from collections import defaultdict, Counter
import json

class SimpleSIEM:
    def __init__(self, loki_url="http://localhost:3100"):
        self.loki_url = loki_url
        self.alerts = []
        self.alert_cooldown = {}  # Prevent alert spam
        
    def query_loki(self, query, minutes=5):
        """Query Loki for log entries"""
        url = f"{self.loki_url}/loki/api/v1/query_range"
        
        end = datetime.utcnow()
        start = end - timedelta(minutes=minutes)
        
        params = {
            'query': query,
            'start': int(start.timestamp() * 1e9),
            'end': int(end.timestamp() * 1e9),
            'limit': 5000
        }
        
        try:
            response = requests.get(url, params=params, timeout=10)
            if response.status_code == 200:
                data = response.json()
                results = []
                for stream in data.get('data', {}).get('result', []):
                    for entry in stream.get('values', []):
                        results.append({
                            'timestamp': int(entry[0]),
                            'line': entry[1],
                            'labels': stream.get('stream', {})
                        })
                return results
        except Exception as e:
            print(f"Error querying Loki: {e}")
        
        return []
    
    def create_alert(self, severity, title, description, **metadata):
        """Create a security alert"""
        alert_key = f"{title}:{metadata.get('source_ip', '')}"
        
        # Check cooldown (don't alert same thing within 10 minutes)
        if alert_key in self.alert_cooldown:
            last_alert = self.alert_cooldown[alert_key]
            if (datetime.utcnow() - last_alert).seconds < 600:
                return
        
        alert = {
            'timestamp': datetime.utcnow().isoformat(),
            'severity': severity,
            'title': title,
            'description': description,
            **metadata
        }
        
        self.alerts.append(alert)
        self.alert_cooldown[alert_key] = datetime.utcnow()
        self.display_alert(alert)
    
    def display_alert(self, alert):
        """Display alert to console"""
        colors = {
            'CRITICAL': '\033[91m',  # Red
            'HIGH': '\033[93m',      # Yellow
            'MEDIUM': '\033[94m',    # Blue
            'LOW': '\033[92m',       # Green
            'END': '\033[0m'
        }
        
        color = colors.get(alert['severity'], '')
        print(f"\n{'='*80}")
        print(f"{color}🚨 [{alert['severity']}] {alert['title']}{colors['END']}")
        print(f"⏰ Time: {alert['timestamp']}")
        print(f"📝 Description: {alert['description']}")
        
        for key, value in alert.items():
            if key not in ['timestamp', 'severity', 'title', 'description']:
                print(f"   {key}: {value}")
        
        print('='*80)
    
    def detect_brute_force(self):
        """Detect brute force login attempts"""
        print("🔍 Checking for brute force attacks...")
        
        # Query failed login attempts
        query = '{job="auth"} |= "Failed" |= "password"'
        events = self.query_loki(query, minutes=10)
        
        # Group by IP
        ip_counts = Counter()
        ip_users = defaultdict(set)
        
        for event in events:
            line = event['line'].lower()
            # Simple IP extraction
            parts = line.split()
            for part in parts:
                if '.' in part and len(part.split('.')) == 4:
                    try:
                        # Validate IP
                        octets = part.split('.')
                        if all(0 <= int(o) <= 255 for o in octets):
                            ip_counts[part] += 1
                            # Try to extract username
                            if 'user' in line:
                                user_idx = parts.index('user') if 'user' in parts else -1
                                if user_idx > 0 and user_idx < len(parts) - 1:
                                    ip_users[part].add(parts[user_idx + 1])
                    except (ValueError, IndexError):
                        pass
        
        # Alert on suspicious activity
        for ip, count in ip_counts.items():
            if count >= 5:
                self.create_alert(
                    severity='HIGH' if count >= 10 else 'MEDIUM',
                    title='Brute Force Attack Detected',
                    description=f"{count} failed login attempts from {ip}",
                    source_ip=ip,
                    failed_attempts=count,
                    targeted_users=list(ip_users[ip]),
                    recommendation='Block this IP address'
                )
    
    def detect_sql_injection(self):
        """Detect SQL injection attempts"""
        print("🔍 Checking for SQL injection attempts...")
        
        # Common SQL injection patterns
        patterns = ["'", "OR 1=1", "UNION SELECT", "DROP TABLE", "--", ";"]
        
        query = '{job=~"nginx|api"}'
        events = self.query_loki(query, minutes=15)
        
        injection_attempts = defaultdict(int)
        
        for event in events:
            line = event['line']
            for pattern in patterns:
                if pattern.lower() in line.lower():
                    # Extract IP if possible
                    parts = line.split()
                    for part in parts:
                        if '.' in part:
                            try:
                                octets = part.split('.')
                                if len(octets) == 4 and all(0 <= int(o) <= 255 for o in octets):
                                    injection_attempts[part] += 1
                            except (ValueError, IndexError):
                                pass
        
        for ip, count in injection_attempts.items():
            if count >= 3:
                self.create_alert(
                    severity='CRITICAL',
                    title='SQL Injection Attack Detected',
                    description=f"{count} SQL injection patterns from {ip}",
                    source_ip=ip,
                    attack_count=count,
                    recommendation='Block this IP and review WAF rules'
                )
    
    def detect_unusual_traffic_spike(self):
        """Detect unusual traffic spikes"""
        print("🔍 Checking for traffic spikes...")
        
        # Get current traffic
        query = 'sum(count_over_time({job=~"nginx|api"}[5m]))'
        current_count = 0
        
        try:
            url = f"{self.loki_url}/loki/api/v1/query"
            params = {'query': query}
            response = requests.get(url, params=params, timeout=10)
            if response.status_code == 200:
                data = response.json()
                result = data.get('data', {}).get('result', [])
                if result:
                    current_count = float(result[0]['value'][1])
        except:
            return
        
        # Simple threshold (in production, use statistical methods)
        if current_count > 1000:  # More than 1000 requests in 5 minutes
            self.create_alert(
                severity='MEDIUM',
                title='Traffic Spike Detected',
                description=f"Unusual traffic: {current_count:.0f} requests in 5 minutes",
                request_count=int(current_count),
                recommendation='Check for DDoS attack or legitimate traffic surge'
            )
    
    def detect_error_surge(self):
        """Detect sudden increase in errors"""
        print("🔍 Checking for error surges...")
        
        query = 'sum(rate({level="ERROR"}[5m]))'
        
        try:
            url = f"{self.loki_url}/loki/api/v1/query"
            params = {'query': query}
            response = requests.get(url, params=params, timeout=10)
            if response.status_code == 200:
                data = response.json()
                result = data.get('data', {}).get('result', [])
                if result:
                    error_rate = float(result[0]['value'][1])
                    if error_rate > 1:  # More than 1 error per second
                        self.create_alert(
                            severity='HIGH',
                            title='Error Rate Surge',
                            description=f"High error rate: {error_rate:.2f} errors/second",
                            error_rate=error_rate,
                            recommendation='Check application logs for root cause'
                        )
        except Exception as e:
            pass
    
    def generate_report(self):
        """Generate summary report"""
        if not self.alerts:
            return
        
        print(f"\n{'='*80}")
        print(f"📊 SECURITY SUMMARY REPORT")
        print(f"{'='*80}")
        print(f"Total Alerts: {len(self.alerts)}")
        
        by_severity = Counter(a['severity'] for a in self.alerts)
        for severity in ['CRITICAL', 'HIGH', 'MEDIUM', 'LOW']:
            if severity in by_severity:
                print(f"  {severity}: {by_severity[severity]}")
        
        print(f"{'='*80}\n")
    
    def run(self, interval=60):
        """Run SIEM monitoring loop"""
        print(f"🛡️  Starting Simple SIEM System")
        print(f"📡 Monitoring Loki at: {self.loki_url}")
        print(f"⏱️  Check interval: {interval} seconds")
        print(f"{'='*80}\n")
        
        iteration = 0
        try:
            while True:
                iteration += 1
                print(f"\n[{datetime.utcnow().isoformat()}] Running security checks (iteration {iteration})...")
                
                self.detect_brute_force()
                self.detect_sql_injection()
                self.detect_unusual_traffic_spike()
                self.detect_error_surge()
                
                if iteration % 10 == 0:  # Report every 10 iterations
                    self.generate_report()
                
                print(f"✅ Checks complete. Sleeping for {interval} seconds...")
                time.sleep(interval)
                
        except KeyboardInterrupt:
            print("\n🛑 Stopping SIEM...")
            self.generate_report()
        except Exception as e:
            print(f"❌ Error: {e}")

if __name__ == "__main__":
    siem = SimpleSIEM()
    siem.run(interval=60)
```
